	SUBROUTINE MINIMU(IMINUTI,IVET,I1,I2,MINI,J)
	calcola il valore minimo del vettore ivet dall'indice I1
	all'indice I2 restituendo il numero di valori validi trovati in J
	solo nel caso che l'intervallo non sia superiore a INTMAX
	(default=2 ore) altrimenti MAXI=32767 ; J=-1

	INPUT:

	IMINUTI(I2)	I*4	vettore dei tempi dei dati	
	IVET(I2)	I*4	vettore dei dati (dato mancante =32767)
	I1		I*4	indice del vettore IVET da cui iniziare la
				valutazione
	I2		I*4	indice del vettore IVET in cui finire la
				valutazione

	OUTPUT:

	MINI		R*4	valore minimo trovato
				se :
				intervallo tra i dati >INTMAX MAXI=32767
				dati elaborati        =0      MAXI=32767
	J		I*4	numero di dati effettivi su cui e` stato
				valutato il valore minimo.
				se :
				intervallo tra i dati >INTMAX j=-1
