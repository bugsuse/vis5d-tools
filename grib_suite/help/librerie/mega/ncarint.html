	SUBROUTINE NCARINT(S,INIZ,LUN,IER)

	VERIFICA CHE IL CONTENUTO DI UNA STRINGA DI CARATTERI
	SIA SOLO NUMERICO.
	sono validi solo (1;2;3;4;5;6;7;8;9;0)

	IN:
	S	CHAR*		STRINGA DA ESAMINARE
	INIZ	I*4		POSIZIONE INIZIALE DA ESAMINARE
	LUN	I*4		NUMERO CARATTERI DA ESAMINARE

	OUT:
	IER	I*4		=0 TUTTO O.K.
				=1 SONO PRESENTI CARATTERI NON NUMERICI

Torna all'indice
